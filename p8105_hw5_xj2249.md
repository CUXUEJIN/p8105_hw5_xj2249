# write a function
replace_missing <- function(vector) {
        
        if (is.numeric(vector)) {
                replace_na(vector,mean(vector,na.rm = TRUE))
        } else if (is.character(vector)) {
                replace_na(vector,"virginica")
        }
}

# apply my 'replace_missing' function
iris_without_missing <-
        iris_with_missing %>% 
        map(replace_missing) %>% 
        bind_cols() %>% 
        janitor::clean_names()
